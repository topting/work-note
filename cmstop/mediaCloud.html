    # 管理后台

    # 一、资源加载规范

    • 利用 RequireJS 库，遵循 AMD 规范，配合 Anaular 的路由，进行模块化管理。 路由的开始第一个字符串，为一个模块开始。
    • 根据Angular 的路由，先加载依赖文件（loaDeps），在加载本模块的 js文件 （$.getScript）。
    $routeProvider.when('/:module/:controller/:action', {
    templateUrl: function(params) {
        return '/assets/module/' + params.module + '/' + params.controller + '/template/' + params.action + '.html';
    },
    resolve: {
        /**
            * 动态处理依赖
            *
            * 按照路由从 {module}/{controller}/deps/{action}.json 中获取依赖列表
            * 定义 {module}-{controller}-{action}-controller 的控制器并
            * 自动加载 {module}/{controller}/js/{action}.js
            */
        load: ['$q', '$route', '$rootScope', '$location', '$templateCache',
            function ($q, $route, $rootScope, $location, $templateCache) {
                var depsDeferred = $q.defer(),
                    params = $route.current.params,
                    dependends = [],
                    baseUrl = '/assets/module/' + params.module + '/' + params.controller + '/',
                    deps;

                return $q(function (resolve, reject) {
                    cloudModule.currentRoute = params;
                    $.getJSON(baseUrl + 'deps/' + params.action + '.json?' + cloud.staticVersion).always(function(data, status, error) {
                        deps = data;
                        if (status !== 'success') {
                            console.report('Parse dependends (.json) failed', params, error);
                            return reject();
                        } else if (status === 'success' && deps) {
                            params = angular.extend(params, deps);
                            deps = deps.dependends ? deps.dependends : [];
                            resolve();
                        }
                    });
                }).then(function () {
                    return $q(function (resolve, reject) {
                        if (cloud.usePack) {
                            $.getJSON('/assets/pack/'+params.module+'/template.json', function (data) {
                                angular.forEach(data, function (data, path) {
                                    $templateCache.put(path, data);
                                    $templateCache.put(path + '?' + cloud.staticVersion, data);
                                    resolve();
                                })
                            })
                        } else {
                            resolve();
                        }
                    });
                }).then(function () {
                    return loadDeps(deps, params);
                }).then(function () {
                    return $.getScript(baseUrl + 'js/' + params.action + '.js?' + cloud.staticVersion);
                }).catch(function(result, state, error) {
                    if (state === 'error') {
                        if (result.status === 404) {
                            return $location.url('index/index/404');
                        }
                        console.report('Load controller failed', params, error, this.url);
                    } else if (state === 'parsererror') {
                        console.report('Controller Parse Error', params, error, this.url);
                    } else {
                        console.report('Controller Parse Error', params, '', this.url);
                    }
                });
            }
        ]
    }
})
.when('/app/:app', {
    // ...
})
.when('/', {
    redirectTo: '/index/index/index'
}).otherwise({
    redirectTo: '/index/index/404'
});

    # 三、权限配置

    • 权限总体分为 模块权限 和 操作权限。
    • 前端模块与接口形成一一对应关系。每个操作，页面对应一个接口的 action方法。在用户登录后，接口返回当前用户的所有权限模块、操作权限。

    # 模块权限判断

    全局变量的 window.cloud.app_group、window.cloud.app、window.cloud.appAlias，存有当前登录用户的所有权限模块。在路由变化时，先判断是否有拥有模块权限。若没有，则跳转至首页。

    # 操作权限判断

    操作权限为一个字符串，前端通过 aca 方法，判断用户是否拥有操作权限。

    $rootScope.aca = function (ca, app) {
    app = app || cloudModule.currentRoute.module;
    if (cloud.aca.su) {
        return true;
    }
    if (cloud.aca[app] && cloud.aca[app].indexOf(ca) > -1) {
        return true;
    }
    return false;
};

    # 五、稿件编辑器

    # 自定义功能

    选取系统图片、音频、视频等素材 以下以插入音频素材为例

    1. 读取配置 config.toolbars，遍历工具栏数据，若是自定义按钮，则进行对应的初始化操作。
    2. 初初始化自定义操作为一个 UE 的 button，并且给 button 绑定点击事件。事件主要进行调用系统素材的逻辑（打开素材选取界面），插入素材的方法（window.audioUpload）。
    3. 选取素材后，调用上一步中的 window.audioUpload，向编辑器中插入音频信息。window.audioUpload 中调用 UEditor 的插入音频事件。

    # 接入第三方编辑器

    接入秀米编辑器、135编辑器 以下为接入 【秀米编辑器】为例

    1. config.toolbars 中配置秀米的操作按钮，并在 UEditor 加载前，引入 秀米相关的 js 、css等（define ）方法
    2. 初始化加载编辑器时，执行秀米编辑器对应的  js 方法：cloud.UE.loadModule.xiumi($scope.id)
    3. cloud.UE.loadModule.xiumi($scope.id) 执行时，按照 UEditor 的规范，注册 Dialog 对象。对象注册成功后，实例化一个 Button ，对应编辑器中的秀米按钮。
    4. 加载秀米的 html 页面中，监听秀米的 mesaage 事件，并将编辑好的内容插入 Ueditor 中

    示例代码：

    注册 dialog

    cloud.UE.loadModule.xiumi = function(id) {
        UE.registerUI('dialog' + id, function (editor, uiName) {
            var btn = new UE.ui.Button({
                name   : 'xiumi-connect',
                title  : '秀米',
                onclick: function () {
                    var dialog = new UE.ui.Dialog({
                        iframeUrl: '/assets/cloud/ui/ueditor/module/xiumi/xiumi-ue-dialog-v5.html',
                        editor   : editor,
                        name     : 'xiumi-connect',
                        title    : "秀米图文消息助手",
                        cssRules : "width: " + (window.innerWidth - 60) + "px;" + "height: " + (window.innerHeight - 60) + "px;",
                    });
                    dialog.render();
                    dialog.open();
                }
            });

            return btn;
}, 104, id);

    插入编辑内容

    window.addEventListener('message', function (event) {
    if (event.origin == xiumi_url) {
        editor.execCommand('insertHtml', event.data);
        editor.fireEvent('afterpaste');// 触发图片转存逻辑
        dialog.close();
    }
}, false);
